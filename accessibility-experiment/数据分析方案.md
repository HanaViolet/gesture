#### 效果量计算
```python
def cohens_d(group1, group2):
    """计算Cohen's d效应量"""
    n1, n2 = len(group1), len(group2)
    pooled_std = np.sqrt(((n1-1)*group1.var() + (n2-1)*group2.var()) / (n1+n2-2))
    return (group1.mean() - group2.mean()) / pooled_std

def interpret_cohens_d(d):
    """解释效应量大小"""
    abs_d = abs(d)
    if abs_d < 0.2:
        return "可忽略"
    elif abs_d < 0.5:
        return "小效应"
    elif abs_d < 0.8:
        return "中效应"
    else:
        return "大效应"
```

```python
def calculate_weighted_nasa_tlx(row):
    """计算加权NASA-TLX得分"""
    dimensions = ['mental', 'physical', 'temporal', 'performance', 'effort', 'frustration']
    weights = [row[f'w_{d}'] for d in dimensions]
    scores = [row[d] for d in dimensions]

    # performance反向计分
    scores[3] = 100 - scores[3]

    weighted_sum = sum(w * s for w, s in zip(weights, scores))
    return weighted_sum / sum(weights)

# 或使用未加权版本（推荐，更稳定）
def calculate_raw_nasa_tlx(row):
    """计算原始NASA-TLX得分"""
    scores = [row[d] for d in ['mental', 'physical', 'temporal', 'performance', 'effort', 'frustration']]
    scores[3] = 100 - scores[3]  # performance反向
    return np.mean(scores)
```

#### SUS得分计算
```python
def calculate_sus(scores):
    """
    计算SUS得分
    scores: [q1, q2, ..., q10] 1-5分
    """
    # 奇数题：得分-1
    # 偶数题：5-得分
    converted = []
    for i, score in enumerate(scores):
        if i % 2 == 0:  # 奇数题（0-indexed）
            converted.append(score - 1)
        else:  # 偶数题
            converted.append(5 - score)

    total = sum(converted) * 2.5
    return total

# 解释SUS得分
def interpret_sus(score):
    if score >= 80.3:
        return "优秀"
    elif score >= 68:
        return "可用"
    elif score >= 51:
        return "边缘"
    else:
        return "不可用"
```

### 6.2 数据质量控制
```python
def data_quality_check(df):
    """数据质量检查"""
    issues = []

    # 缺失值检查
    missing = df.isnull().sum()
    if missing.any():
        issues.append(f"缺失值发现: {missing[missing > 0].to_dict()}")

    # 异常值检查
    for col in ['completion_time', 'sus_score']:
        if col in df.columns:
            q1, q3 = df[col].quantile([0.25, 0.75])
            iqr = q3 - q1
            outliers = df[(df[col] < q1 - 1.5*iqr) | (df[col] > q3 + 1.5*iqr)]
            if len(outliers) > 0:
                issues.append(f"{col}异常值: {len(outliers)}个")

    # 逻辑一致性检查
    if 'success' in df.columns and 'completion_time' in df.columns:
        inconsistent = df[(df['success'] == False) & (df['completion_time'] < 5)]
        if len(inconsistent) > 0:
            issues.append(f"逻辑不一致: {len(inconsistent)}条记录失败但时间<5秒")

    return issues
```
